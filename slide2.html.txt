function getFilteredData() {
  if (activeDivision === 'TODAS') return rentData;

function formatMonto(v) {
  if (v >= 1000000) return 'S/ ' + (v/1000000).toFixed(2) + 'M';
  if (v >= 1000) return 'S/ ' + (v/1000).toFixed(1) + 'K';
  return 'S/ ' + v.toFixed(0);
}

// DIVISION FILTERS
function renderDivisionFilters() {
  var divs = [];
  rentData.forEach(function(d) {
    if (divs.indexOf(d.division) === -1) divs.push(d.division);
  });
  var container = document.getElementById('divFilters');
  var html = '';
  html += '<div class="div-chip active" style="background: linear-gradient(135deg, #38BDF8, #00C9A7); color: #fff;" data-div="TODAS">TODAS</div>';
  divs.forEach(function(div) {
    var color = divisionColors[div.toUpperCase()] || '#38BDF8';
    html += '<div class="div-chip" style="background: ' + color + '20; color: ' + color + '; border: 1px solid ' + color + '40;" data-div="' + div + '">' + div + '</div>';
  });
  container.innerHTML = html;

  // Event listeners para chips
  container.querySelectorAll('.div-chip').forEach(function(chip) {
    chip.addEventListener('click', function() {
      filterDivision(chip.dataset.div);
    });
  });
}

function filterDivision(div) {
  activeDivision = div;
  isTotalGPSMode = false;
  var chips = document.querySelectorAll('.div-chip');
  chips.forEach(function(c) {
    c.classList.toggle('active', c.dataset.div === div);
  });
  updateDonutCenter(div);
  updateDonut();
  renderRanking();
  closeDetail();
}

function updateDonutCenter(div) {
  var big = document.getElementById('donutBig');
  var small = document.getElementById('donutSmall');
  var micro = document.getElementById('donutMicro');
  var totalBig = document.getElementById('centerBig');
  var totalSmall = document.getElementById('centerSmall');

  if (div === 'TODAS') {
    big.textContent = '14.18%';
    small.textContent = 'Total GPS';
    micro.textContent = '22 empresas';
    totalBig.textContent = '14.18%';
    totalSmall.textContent = 'S/ 125,232,042';
  } else {
    var pct = divisionAvgPct[div] || 0;
    var filtered = getFilteredData();
    var totalMonto = filtered.reduce(function(s, d) { return s + d.monto_2025; }, 0);
    big.textContent = pct.toFixed(2) + '%';
    small.textContent = div;
    micro.textContent = filtered.length + ' empresas';
    totalBig.textContent = pct.toFixed(2) + '%';
    totalSmall.textContent = formatMonto(totalMonto);
  }
}

// DONUT CHART
function initDonut() {
  var ctx = document.getElementById('donutChart').getContext('2d');
  var data = getFilteredData();

  var labels = data.map(function(d) { return d.empresa; });
  var values = data.map(function(d) { return d.pct_2025; });
  var colors = data.map(function(d) { return d.color; });

  donutChart = new Chart(ctx, {
    type: 'doughnut',
    data: {
      labels: labels,
      datasets: [{
        data: values,
        backgroundColor: colors,
        borderColor: '#0F172A',
        borderWidth: 3,
        hoverBorderWidth: 5,
        hoverBorderColor: '#fff'
      }]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      cutout: '60%',
      plugins: {
        legend: { display: false },
        tooltip: {
          backgroundColor: 'rgba(15, 23, 42, 0.95)',
          titleColor: '#38BDF8',
          bodyColor: '#E2E8F0',
          borderColor: 'rgba(56, 189, 248, 0.3)',
          borderWidth: 1,
          padding: 14,
          callbacks: {
            label: function(context) {
              var idx = context.dataIndex;
              var d = data[idx];
              return d.empresa + ': ' + d.pct_2025.toFixed(2) + '%';
            }
          }
        }
      },
      onClick: function(e, elements) {
        if (elements.length > 0) {
          var idx = elements[0].index;
          selectEmpresa(data[idx].empresa);
        }
      },
      onHover: function(e, elements) {
        e.native.target.style.cursor = elements.length > 0 ? 'pointer' : 'default';
      }
    }
  });
}

function updateDonut() {
  if (!donutChart) return;
  var data = getFilteredData();
  donutChart.data.labels = data.map(function(d) { return d.empresa; });
  donutChart.data.datasets[0].data = data.map(function(d) { return d.pct_2025; });
  donutChart.data.datasets[0].backgroundColor = data.map(function(d) { return d.color; });
  donutChart.update();
  document.getElementById('donutSubtitle').textContent = activeDivision === 'TODAS' ? '2025' : activeDivision;
}

    html += '<div class="ranking-item ' + isActive + '" data-empresa="' + d.empresa + '">' +
      '<div class="rank ' + isTop3 + '">' + (i + 1) + '</div>' +
      '<div class="info"><div class="name">' + d.empresa + '</div><div class="division">' + d.division + '</div></div>' +
      '<div class="bar-bg"><div class="bar-fill" style="width: ' + width + '%; background: ' + d.color + ';"></div></div>' +
      '<div class="pct" style="color: ' + d.color + ';">' + d.pct_2025.toFixed(2) + '%</div>' +
      '</div>';
  });
  container.innerHTML = html;

  // Event listeners para ranking items
  container.querySelectorAll('.ranking-item').forEach(function(item) {
    item.addEventListener('click', function() {
      selectEmpresa(item.dataset.empresa);
    });
  });
}

// SELECT EMPRESA
function selectEmpresa(empresa) {
  selectedEmpresa = empresa;
  isTotalGPSMode = false;
  var d = null;
  for (var i = 0; i < rentData.length; i++) {
    if (rentData[i].empresa === empresa) { d = rentData[i]; break; }
  }
  if (!d) return;

  var items = document.querySelectorAll('.ranking-item');
  items.forEach(function(item) {
    item.classList.toggle('active', item.dataset.empresa === empresa);
  });

  document.getElementById('donutBig').textContent = d.pct_2025.toFixed(2) + '%';
  document.getElementById('donutSmall').textContent = d.empresa;
  document.getElementById('donutMicro').textContent = d.division;

  var filtered = getFilteredData();
  var rank = 0;
  for (var j = 0; j < filtered.length; j++) {
    if (filtered[j].empresa === empresa) { rank = j + 1; break; }
  }

  showDetailCard(d.empresa, d.division, d.color, [
    { label: '% Margen 2025', value: d.pct_2025.toFixed(2) + '%', colorClass: d.pct_2025 > 0 ? 'positive' : 'negative' },
    { label: 'Monto 2025', value: formatMonto(d.monto_2025), colorClass: 'neutral' },
    { label: 'Ranking', value: '#' + rank, colorClass: 'neutral' },
    { label: 'Division', value: d.division, colorClass: 'neutral' }
  ]);

  renderEvolucion(d.evolucion, d.color, d.empresa);
}

function renderEvolucion(evolucionData, color, title) {
  var ctx = document.getElementById('evolucionChart').getContext('2d');
  var years = ['2018','2019','2020','2021','2022','2023','2024','2025'];
  var pcts = years.map(function(y) {
    return evolucionData[y] && evolucionData[y].pct !== null ? evolucionData[y].pct : null;
  });
  var montos = years.map(function(y) {
    return evolucionData[y] && evolucionData[y].monto !== null ? evolucionData[y].monto / 1000000 : null;
  });

  if (evolucionChart) {
    evolucionChart.destroy();
    evolucionChart = null;
  }

  evolucionChart = new Chart(ctx, {
    type: 'line',
    data: {
      labels: years,
      datasets: [
        {
          label: '% Margen Comercial',
          data: pcts,
          borderColor: color,
          backgroundColor: color + '20',
          fill: true,
          tension: 0.4,
          pointRadius: 6,
          pointHoverRadius: 10,
          pointBackgroundColor: color,
          pointBorderColor: '#fff',
          pointBorderWidth: 2,
          yAxisID: 'y'
        },
        {
          label: 'Monto (Millones S/)',
          data: montos,
          borderColor: '#38BDF8',
          backgroundColor: 'transparent',
          borderDash: [5, 5],
          tension: 0.4,
          pointRadius: 4,
          pointHoverRadius: 8,
          yAxisID: 'y1'
        }
      ]
    },
    options: {
      responsive: true,
      maintainAspectRatio: false,
      interaction: { mode: 'index', intersect: false },
      plugins: {
        legend: {
          labels: { color: '#94A3B8', font: { size: 12 } }
        },
        tooltip: {
          backgroundColor: 'rgba(15, 23, 42, 0.95)',
          titleColor: '#38BDF8',
          bodyColor: '#E2E8F0',
          borderColor: 'rgba(56, 189, 248, 0.3)',
          borderWidth: 1,
          padding: 12
        }
      },
      scales: {
        x: {
          ticks: { color: '#94A3B8' },
          grid: { color: 'rgba(255,255,255,0.05)' }
        },
        y: {
          type: 'linear',
          display: true,
          position: 'left',
          ticks: { color: color, callback: function(v) { return v + '%'; } },
          grid: { color: 'rgba(255,255,255,0.05)' },
          title: { display: true, text: '% Margen', color: color }
        },
        y1: {
          type: 'linear',
          display: true,
          position: 'right',
          ticks: { color: '#38BDF8', callback: function(v) { return 'S/' + v.toFixed(1) + 'M'; } },
          grid: { drawOnChartArea: false },
          title: { display: true, text: 'Monto (M)', color: '#38BDF8' }
        }
      }
    }
  });
}
